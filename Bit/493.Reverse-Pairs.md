## 493. Reverse Pairs

----
## 题目地址

https://leetcode.com/problems/reverse-pairs/

## 题目描述
```
Given an array nums, we call (i, j) an important reverse pair if i < j and nums[i] > 2*nums[j].

You need to return the number of important reverse pairs in the given array.

Example1:

Input: [1,3,2,3,1]
Output: 2
Example2:

Input: [2,4,3,5,1]
Output: 3
Note:
The length of the given array will not exceed 50,000.
All the numbers in the input array are in the range of 32-bit integer.
```

## 代码

### Approach #1 Merge Sort

```java
class Solution {
  public int reversePairs(int[] nums) {
		return mergeSort(nums, 0, nums.length - 1);
  }
  
  private int mergeSort(int[] nums, int s, int e) {
    if (s >= e) return 0;
    int mid = s + (e - s) / 2;
    int cnt = mergeSort(nums, s, mid) + mergeSort(nums, mid + 1, e);
    int j = mid + 1;
    for (int i = s; i <= mid; i++) {
      while (j <= e && nums[i]  > nums[j] * 2) {
        j++;
      }
      cnt += j - (mid + 1);
    }
    
    Arrays.sort(nums, s, e + 1);
    return cnt;
  }
}
```

### Approach #2 BIT

```JAVA
class ReverseNode {
  private int search(int[] bit, int i) {
    int sum = 0;
    while (i < bit.length) {
      sum += bit[i];
      i += i & -i;
    }
    return sum;
  }
  
  private void insert(int[] bit, int i) {
    while (i > 0) {
      bit[i] += 1; // 比bit[i]大的个数
      i -= i & -i;
    }
  }
  
	public int reversePairs(int[] nums) {
    int res = 0;
    int[] copy = Arrays.copyOf(nums, nums.length);
    int[] bit = new int[copy.length + 1];
    Arrays.sort(copy);
    
    for (int ele: nums) {
      // 利用BIT数据结构统计大于2*ele的元素个数
      res += search(bit, index(copy, 2L * ele + 1));
      // 插入当前元素，这里index是ele在copy中的下标
      insert(bit, index(copy, ele));
    }
    return res;
  }
  
  // 返回大于或者等于val的坐标，如果多个val返回第一个坐标
  private int index(int[] nums, long val) {
    int l = 0;
    int r = nums.length - 1;
    while (l <= r) {
      int m = l + ((r - l) >> 1);
      if (nums[m] >= val) {
        r = m - 1;
      } else {
        l = m + 1;
      }
    }
    // bit数组是以1为开始，所以返回l+1
    return l + 1;
  }
}
```



### Approach #3 BST

可惜的是，我们自定义的树并不是一个平衡二叉树，在最坏的情况下会导致长链表，如果LeetCode上使用上述代码会导致TLE(Time Limit Exceeded)。

```JAVA
class ReverseNode {
  public int val;
  public int cnt;
  ReverseNode left;
  ReverseNode right;
  
  public ReverseNode(int val) {
    this.val = val;
    this.cnt = 1;
  }
}

class Solution {
  public int reversePairs(int[] nums) {
    int res = 0;
    ReverseNode root = null;
    for (int ele: nums) {
      res += search(root, 2L * ele + 1);
      root = insert(root, ele); // root.val == ele
    }
    return res;
  }
  
  private int search(ReverseNode root, long val) {
    if (root == null)		return 0;
    if (val == root.val)	return root.cnt;
    if (val < root.val) { // 返回 >= val 的node个数
      return root.cnt + search(root.left, val);
    } else {
      // val > root.val，往root.right中寻找比val大的node
      return search(root.right, val);
    }
  }
  
  private ReverseNode insert(ReverseNode root, int val) {
		if (root == null)		return new ReverseNode(val);
    if (val == root.val) {
      root.cnt++;
      return root;
    } else if (val > root.val) {
      root.cnt++;
      return insert(root.right, val);
    } else {
      return insert(root.left, val);
    }
  }
}
```
